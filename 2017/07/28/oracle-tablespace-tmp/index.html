Oracle tablespace_tmp create

2017-07-28

tablespace_tmp oracle create

//创建临时表空间
create temporary tablespace sdxcpt_temp
tempfile ‘D:/app/Administrator/product/11.2.0/dbhome_1/oradata/sample/sdxcpt_temp.dbf’
size 32m
autoextend on
next 32m maxsize 2048m
extent management local;
//创建数据表空间
create tablespace sdxcpt_date
logging
datafile ‘D:/app/Administrator/product/11.2.0/dbhome_1/oradata/sample/sdxcpt_date.dbf’
size 32m
autoextend on
next 32m maxsize 2048m
extent management local;
//创建用户并指定表空间
create user sdxcpt identified by sdxcpt
default tablespace sdxcpt_date
temporary tablespace sdxcpt_temp;
//给用户授予权限
grant connect,resource to sdxcpt;
